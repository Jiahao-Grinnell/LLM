source .env
docker compose -f compose.yaml up -d  # starts ollama (app is profile-gated)

docker compose -f compose.yaml --profile manual run --rm app \
  --in_dir /data/input \
  --out_dir /data/output \
  --text_out_dir /data/text_out \
  --log_dir /data/logs
docker compose -f compose.yaml down


#or if run ocr model
# Run OCR by overriding entrypoint so Dockerfile ENTRYPOINT can't swallow your args
docker compose run --rm ocr_app \
  --model "glm-ocr:latest" \
  --in_dir /data/input_images \
  --out_dir /data/output_text \
  --ollama_url "http://ollama:11434/api/generate"
